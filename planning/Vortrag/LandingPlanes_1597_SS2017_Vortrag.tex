\begin{frame}{Funktionsaufrufe in Single-Threaded-Umgebungen}{``Synchrone Programmierung''}
	
	\begin{columns}
		\column{0.5\textwidth}
		\begin{itemize}[<+->]
			\item Synchrone Funktionen
			\begin{itemize}[<.->]
				\item werden aufgerufen
				\item berechnen ihr Ergebnis
				\item geben das Resultat zurück
			\end{itemize}
			\item Funktionsaufrufe können die Ausführung blockieren
			\item Das Warten auf externe Ereignisse führt zu Ineffizienz
			\item Es ist keine Neben- läufigkeit möglich
		\end{itemize}
		\column{0.5\textwidth}
			\begin{figure}
				\includegraphics[height=.8\textheight]{images/syncFlow.png}
			\end{figure}
	\end{columns}

	\note{
		\begin{itemize}
			\item Das Hauptprogramm kann nicht weiterarbeiten
			\item Nebenläufigkeit ist nicht möglich
			
			\skippingparagraph
			
			\skippingparagraph
			
			\item Was lässt sich daran besser machen?
		\end{itemize}
	}

\end{frame}

	\begin{frame}{Asynchrone Programmierung}
		
	\begin{itemize}[<+->]
		\item<1-> Asynchrone Funktionen verhalten sich anders:	
		\begin{itemize}[<+->]
			\item<+-> Async Funktionen werden \alert{jetzt} aufgerufen, \ldots
			\item<+-> \ldots kehren direkt zurück\ldots
			\item<+-> \ldots und berechnen ihr Ergebnis
			\item<+-> Diese Ergebnis wird \alert{später} übergeben
		\end{itemize}
		\item<+-> Der aufrufende Code erhält die Kontrolle direkt zurück
		\item<+-> Das Hauptprogramm kann sinnvolle Dinge tun, während die async Funktion noch ihr Ergebnis ermittelt
		\item<+-> Dadurch gibt es \alert{Nebenläufigkeit} auch in single-threaded Umgebungen
	\end{itemize}

	\note{
		Erkläre hier, dass die asynchronen Funktionen eben \alert{doch in einem anderen Thread} ablaufen können.

		Eventuell sogar noch die Grafik Zeigen:

			\includegraphics[height=.5\textheight]{images/syncFlow.png}
			\caption{Asnychronous Functions are executed outside the JavaScript-Single-Thread}
		\end{figure}
		
	}
	\end{frame}
